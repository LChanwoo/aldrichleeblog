### 문제풀이

- 두 포인터를 이용하여 풀 수 있는 문제이다.
- 그러나 이번 문제는 브루트포스로 풀어보았다.

```cpp
#include<iostream>
#include<algorithm>
using namespace std;

int n, k;
long long a[100001], ans;

int main(){
    ios_base::sync_with_stdio(false);
    cin.tie(NULL);
    cout.tie(NULL);
    cin >> n;
    for (int i = 0; i < n;i++) cin >> a[i];
    cin >> k;
    for (int i = 0; i < n; i++)
    {
        long long sum = 0;
        for (int j = i; j < n;j++){
            sum += a[j];
            if(sum>k) ans++;
        }
    }
    cout << ans;
}
```
